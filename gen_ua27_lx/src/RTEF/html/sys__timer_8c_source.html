118static DWORD WINAPI TimerThread(LPVOID lpParam) {
119 uint8_t timer_id = (uint8_t)(uintptr_t)lpParam;
120 LARGE_INTEGER dueTime;
121 dueTime.QuadPart = -(int64_t)(timer_id == TIMER_10ms ? TIMER_PERIOD_10MS :
122 timer_id == TIMER_100ms ? TIMER_PERIOD_100MS :
123 TIMER_PERIOD_200MS) * 10000LL;
124
125 while (1) {
126 SetWaitableTimer(timer_manager[timer_id].handle, &dueTime,
127 timer_id == TIMER_10ms ? TIMER_PERIOD_10MS :
128 timer_id == TIMER_100ms ? TIMER_PERIOD_100MS :
129 TIMER_PERIOD_200MS,
130 NULL, NULL, 0);
131
132 WaitForSingleObject(timer_manager[timer_id].handle, INFINITE);
133 timer_callback_handler(timer_id);
134 }
135 return 0;
136}

138static void timer_callback_handler(uint8_t timer_id) {
139 EnterCriticalSection(&timer_manager[timer_id].lock);
140
141 timer_callback_entry_t *entry = timer_manager[timer_id].callbackList;
142 while (entry) {
143 if (entry->callback && entry->state == ARM) {
144 entry->callback(entry->context);
145 }
146 entry = entry->next;
147 }
148
149 LeaveCriticalSection(&timer_manager[timer_id].lock);
150}

186// Add a callback to a timer with priority
187timer_callback_entry_t* timer_manager_add_callback(uint8_t timer_id, timer_callback_t callback, void *context, uint8_t priority) {
188 if (timer_id >= MAX_TIMERS) {
189 return 0; // Invalid timerId
190 }
191
192 timer_callback_entry_t *newEntry = (timer_callback_entry_t*) malloc(sizeof(timer_callback_entry_t));
193 if (!newEntry) {
194 return NULL; // Memory allocation failed
195 }
196#ifdef _WIN32
197 InitializeCriticalSection(&newEntry->sem_entry);
198#else
199 newEntry->sem_entry = xSemaphoreCreateMutex();
200#endif
201 newEntry->timer_id = timer_id;
202 newEntry->state = DISARM;
203 newEntry->callback = callback;
204 newEntry->context = context;
205 newEntry->priority = priority;
206 newEntry->next = NULL;
207
208 // Insert the new callback in sorted order based on priority
209 timer_callback_entry_t **head = &timer_manager[timer_id].callbackList;
210 if (*head == NULL || (*head)->priority < priority) {
211 // Insert at the head if higher priority
212 newEntry->next = *head;
213 *head = newEntry;
214 } else {
215 // Insert in sorted order
216 timer_callback_entry_t *current = *head;
217 while (current->next != NULL && current->next->priority >= priority) {
218 current = current->next;
219 }
220 newEntry->next = current->next;
221 current->next = newEntry;
222 }
223
224 return newEntry; // Success
225}

227// Remove a callback from a timer
228void timer_manager_remove_callback(uint8_t timer_id, timer_callback_t callback, void *context) {
229 if (timer_id >= MAX_TIMERS)
230 return;
231
232 timer_callback_entry_t **head = &timer_manager[timer_id].callbackList;
233 timer_callback_entry_t *current = *head, *prev = NULL;
234
235 while (current != NULL) {
236 if (current->callback == callback && current->timer_id == timer_id && current->context == context) {
237 if (prev == NULL) {
238 *head = current->next; // Remove first element
239 } else {
240 prev->next = current->next;
241 }
242 free(current);
243 return;
244 }
245 prev = current;
246 current = current->next;
247 }
