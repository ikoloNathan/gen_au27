55static void dispatch(base_obj_t *const me, const message_frame_t *frame) {

167void ao_can_send_with_timeout(can_obj_t *me, const message_frame_t *frame) {
168 if (!me || !me->timerManager)
169 return;
170
171 if (count >= MAX_PENDING_FRAMES) {
172 head = (head + 1) % MAX_PENDING_FRAMES; // Overwrite oldest
173 count--;
174 }
175
176 uint8_t index = tail;
177 memcpy(&pendingFrames[index].frame, frame, sizeof(message_frame_t));
178
179 // Add timeout callback
180 pendingFrames[index].timerEntry = me->timerManager->add_callback(TIMER_100ms, ao_can_timeout, &pendingFrames[index], 1);
181
182 if (pendingFrames[index].timerEntry) {
183 me->timerManager->arm(pendingFrames[index].timerEntry);
184 }
185
186 // Move tail forward (circular buffer logic)
187 tail = (tail + 1) % MAX_PENDING_FRAMES;
188 count++;
189
190 // Send CAN frame
191 llce_can_write(frame->type, *frame);
192}

222void ao_can_clear_timeout(can_obj_t *me, message_frame_t frame) {
223 uint8_t i = head;
224 uint8_t processed = 0;
225
226 while (processed < count) {
227 if ((message_encode_can_id(&pendingFrames[i].frame) == message_encode_can_id(&frame)) && (pendingFrames[i].frame.meta.correlationId == frame.meta.correlationId)) {
228
229 if (pendingFrames[i].timerEntry) {
230 me->timerManager->disarm(pendingFrames[i].timerEntry);
231 me->timerManager->remove_callback(TIMER_100ms, ao_can_timeout, pendingFrames[i].timerEntry->context);
232 pendingFrames[i].timerEntry = NULL;
233 }
234
235 // Shift remaining entries forward
236 uint8_t j = i;
237 while (j != tail) {
238 uint8_t next = (j + 1) % MAX_PENDING_FRAMES;
239 pendingFrames[j] = pendingFrames[next];
240 j = next;
241 }
242
243 // Update tail and count
244 tail = (tail == 0) ? MAX_PENDING_FRAMES - 1 : tail - 1;
245 count--;
246 return;
247 }
248 i = (i + 1) % MAX_PENDING_FRAMES;
249 processed++;
250 }
251}
