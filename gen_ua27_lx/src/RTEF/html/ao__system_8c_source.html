302static void dispatch(base_obj_t *const me, const message_frame_t *frame) {
303 event_t event = { 0 };
304 switch (frame->topicOrServiceId) {
305 case INTERNAL_SYSTEM_TIME:
306#ifdef _WIN32
307 me->last_heartbeat_time = (uint32_t)get_time_ms();
308#else
309 me->last_heartbeat_time = xTaskGetTickCount();
310#endif
311 break;
312 case INTERNAL_SYSTEM_STATE:
313 event.signal = frame->payload[0];
314 fsm_handler(&me->fsm, &event);
315 break;
316 }
317}
