82unsigned __stdcall udp_recv_task(void *param) {
83 Broker_t *broker = (Broker_t*)param;
84 while (1) {
85 int recvLen = recvfrom(Socket, buffer, BUFFER_SIZE, 0,
86 (struct sockaddr*) &clientAddr, &clientAddrLen);
87
88 Message_Frame frame = {0};
89 uint8_t frame_size = recvLen > sizeof(Message_Frame) ? sizeof(Message_Frame) : recvLen ;
90
91 memcpy(&frame,buffer,frame_size);
92
93 memset(buffer,0,BUFFER_SIZE);
94 if(broker != NULL){
95 Broker_Post(broker,frame,frame.priority != 3);
96 }
97 }
98 return 0;
99}
