64static void MsgQueue_Push(MsgQueue_t *q, const message_frame_t *frame) {
65 EnterCriticalSection(&q->lock);
66 if (q->count < AO_QUEUE_SIZE) {
67 q->buffer[q->tail] = *frame;
68 q->tail = (q->tail + 1) % AO_QUEUE_SIZE;
69 q->count++;
70 WakeConditionVariable(&q->cond);
71 }
72 LeaveCriticalSection(&q->lock);
73}

81static int MsgQueue_Pop(MsgQueue_t *q, message_frame_t *frame) {
82 int success = 0;
83 EnterCriticalSection(&q->lock);
84 while (q->count == 0) {
85 SleepConditionVariableCS(&q->cond, &q->lock, INFINITE);
86 }
87 if (q->count > 0) {
88 *frame = q->buffer[q->head];
89 q->head = (q->head + 1) % AO_QUEUE_SIZE;
90 q->count--;
91 success = 1;
92 }
93 LeaveCriticalSection(&q->lock);
94 return success;
95}

161void stop(base_obj_t *const me) {
162 unregister_active_object(me);

167 me->thread_id = NULL;
168 }

170 me->vptr->logger(me, (const uint8_t*)"ActiveObject stopped.",sizeof("ActiveObject stopped."));
171#else
172 if (me->thread_id) {
173 vTaskDelete(me->thread_id);
174 me->thread_id = NULL;
175 }
176 if (me->msg_queue_id) {
177 vQueueDelete(me->msg_queue_id);
178 }

317unsigned __stdcall event_loop(void *vparam) {
318 base_obj_t *me = (base_obj_t*) vparam;
319 message_frame_t event;
320
321 while (1) {
322 if (MsgQueue_Pop(&me->msgQueue, &event)) {
323 if (me->vptr && me->vptr->dispatch) {
324 me->vptr->dispatch(me, &event);
325 memset(&event, 0, sizeof(message_frame_t));
326 }
327 }
328 }
329 return 0;
330}
